int lenline;
float radtodeg = (3.141592/180);

void drawballline()
{
 lenline = (gy/10);
 if (theta>=150)
  theta = 150;
 if (theta<=30)
  theta = 30;
 int x,y;
 y = (gy-34) - lenline*(sin(theta*radtodeg));
 x = gx/2 + lenline*(cos(theta*radtodeg));
 line(gx/2,gy-34,x,y);
}

void choosevel()
{
 vbx = lenline*(cos(theta*radtodeg))/10;
 vby = -lenline*(sin(theta*radtodeg))/10;
 game();
}
